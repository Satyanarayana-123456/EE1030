 \begin{align}
 \vec{A} & = c\myvec{\cos{\brak{\angle B}} \\ \sin{\brak{\angle B}}} \\
 \vec{A} & = \myvec{0 \\ c} \\

 \end{align}

\begin{align}
c & = \frac{\frac{121}{4} - \frac{49}{4}}{2\brak{\frac{11}{2} - 0}} \\
c & = \frac{18}{11}
\end{align}
